import threading
from pynput.keyboard import Key, Listener




class KeyInput:
    def __init__(self):
        thread = threading.Thread(target=self.keyThread, args=(self,))
        thread.start()
        self.scannedValue = None
        self.buffer = str()

    def keyThread(self, arg):
        with Listener(on_press=self.on_press) as listener:
            listener.join()

    def on_press(self, key):
        if key == Key.enter:
            print(self.buffer)
            self.scannedValue = self.buffer
            self.buffer = str()
        elif str(key).replace("'","") == "\\x03":
            print("Break")
            return False
        else:
            try:
                value = str(key).replace("'","")
                if "Key" not in value:
                    self.buffer += value
            except:
                self.buffer = str()

    def read(self):
        value = self.scannedValue
        self.scannedValue = None
        return value

if __name__ == '__main__':
    print("Start key thread");

    keyInput = KeyInput()
